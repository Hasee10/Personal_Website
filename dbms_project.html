Frontend

The frontend is built with React and modern CSS, providing a seamless and interactive user experience for managing and visualizing data. 

// Example React component
function fetchScores(callback) {
  const query = `
      SELECT s.*, e.Event_Name, u.Name as Participant_Name, j.Judge_ID, l.Ranks
      FROM Score s
      JOIN Event e ON s.Event_ID = e.EventID
      JOIN User u ON s.Participant_ID = u.UserID
      JOIN Judge j ON s.Judge_ID = j.Judge_ID
      LEFT JOIN Leaderboard l ON s.Event_ID = l.Event_ID AND s.Participant_ID = l.Participant_ID
      ORDER BY e.Event_Name, l.Ranks`;
   connection.query(query, callback);
}
